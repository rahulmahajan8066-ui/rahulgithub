Sub VariableAdd()
Dim num As Integer
num = Range("f2").Value
If num > 0 Then
Range("g2").Value = "Positive"
Else
Range("g2").Value = "Negative"
End If


End Sub
Sub EvenorOdd()
Dim num As Integer
num = Range("i2").Value
If num Mod 2 = 0 Then
Range("j2").Value = "Given Number is Even"
Else
Range("j2").Value = "Given Number is ODD"
End If

End Sub
Sub MsgBox()
Dim num As Integer
num = Int(InputBox("Enter any number"))
If num Mod 2 = 0 Then
MsgBox (Str(num) + " is even number")
Else
MsgBox (Str(num) + " is odd number")
End If
End Sub
